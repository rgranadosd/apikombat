# All (Fundamental WildCard)
	Targets:
		lambda self, p: p == self.controller

	Play:
		# Seleccionar el primer aspecto disponible
		from MTG.api.aspecto import AspectoPermanent
		aspectos_disponibles = ['seguridad', 'documentacion', 'gobierno', 'performance']
		aspectos_existentes = [
			asp.aspecto_type for asp in self.controller.battlefield
			if hasattr(asp, 'is_aspecto') and asp.is_aspecto
		]
		for tipo in aspectos_disponibles:
			if tipo not in aspectos_existentes:
				aspecto = AspectoPermanent(tipo, controller=self.controller)
				self.controller.battlefield.add(aspecto)
				return True
		return False

			and not p.esta_protegido())

	Play:
		if hasattr(self.controller.opponent, 'code_freeze_shield') and self.controller.opponent.code_freeze_shield:
			self.controller.opponent.code_freeze_shield = False
			return True
		if self.targets_chosen[0].esta_protegido():
			return False
		aspecto = self.targets_chosen[0]
		if aspecto.vulnerable:
			aspecto.controller.battlefield.remove(aspecto)
			aspecto.controller.graveyard.add(aspecto)
		else:
			aspecto.vulnerar()

# Refactoring (Management)
	Targets:
		lambda self, p: (hasattr(p, 'is_aspecto') and p.is_aspecto 
			and p.controller == self.controller)
		lambda self, p: (hasattr(p, 'is_aspecto') and p.is_aspecto 
			and p.controller == self.controller.opponent)

	Play:
		if hasattr(self.controller.opponent, 'code_freeze_shield') and self.controller.opponent.code_freeze_shield:
			self.controller.opponent.code_freeze_shield = False
			return True
		aspecto_mio = self.targets_chosen[0]
		aspecto_oponente = self.targets_chosen[1]
		aspecto_mio.controller.battlefield.remove(aspecto_mio)
		aspecto_oponente.controller.battlefield.remove(aspecto_oponente)
		aspecto_mio.controller = self.controller.opponent
		aspecto_oponente.controller = self.controller
		self.controller.battlefield.add(aspecto_oponente)
		self.controller.opponent.battlefield.add(aspecto_mio)
		return True

				for asp in self.controller.battlefield

		aspecto_objetivo = self.targets_chosen[0]
		oponente = self.controller.opponent
		from MTG.api.aspecto import AspectoPermanent
		nuevo_aspecto = AspectoPermanent(
			aspecto_objetivo.aspecto_type,
			controller=self.controller
		)
		nuevo_aspecto.vulnerable = aspecto_objetivo.vulnerable
		nuevo_aspecto.protecciones = aspecto_objetivo.protecciones
		self.controller.battlefield.add(nuevo_aspecto)
		oponente.battlefield.remove(aspecto_objetivo)
		oponente.graveyard.add(aspecto_objetivo)
		return True

			self.controller.opponent.code_freeze_shield = False
			return True
		aspecto_mio = self.targets_chosen[0]
		tipo_aspecto = aspecto_mio.aspecto_type
		aspecto_oponente = None
		for asp in self.controller.opponent.battlefield:
			if (hasattr(asp, 'is_aspecto') and asp.is_aspecto 
				and asp.aspecto_type == tipo_aspecto
				and not asp.esta_fortalecido()
				and not asp.vulnerable):
				aspecto_oponente = asp
				break
		if aspecto_oponente:
			aspecto_oponente.vulnerar()
			return True
		return False
